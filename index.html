# Soient les points A (xA ; yA) et B (xB ; yB).

x = np.linspace(0, 10, 1000) # Initialisation de l'axe des abscisses sur l'intervalle [0 ; 10]

def borne(b1, b2) : return( (x >= b1) & (x <= b2) ) # Restriction de x sur l'intervalle [b1 ; b2].

def cd(xA, yA, xB, yB) : return((yB - yA) / (xB - xA)) # Retourne le coefficient directeur de la droite passant A et B.

def od(a, xA, yA) : return(yA - a*xA) # Retourne l'ordonne  l'origine de la droite de coefficient directeur a passant par A.

def affine_cd(a, xA, yA) : # Retourne la fonction affine de coefficient directeur a et passant par le point A.
    b = yA - a*xA
    return(a*x + b)

def affine(xA, yA, xB, yB) : # Retourne la fonction affine passant par les points A et B.
    a = (yB - yA) / (xB - xA)
    b = yA - a*xA
    return(a*x + b)

def segment(xA, yA, xB, yB) : # Retourne la fonction affine passant par A et B, restreinte du point A au point B.
    a = (yB - yA) / (xB - xA)
    b = yA - a*xA
    return(a*x[borne(xA, xB)] + b)

def trace(xA, yA, xB, yB) : plt.plot([xA, xB], [yA, yB], color="black") # Trace la droite de A  B.
