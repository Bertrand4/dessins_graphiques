x = np.linspace(0, 10, 1000)

# Soient les points A (xA ; yA) et B (xB ; yB).

def affine(xA, yA, xB, yB) : # Retourne la fonction affine passant par les points A et B.
    a = (yB - yA) / (xB - xA)
    b = yA - a*xA
    return(a*x + b)

def cd(xA, yA, xB, yB) : # Retourne le coefficient directeur de la droite passant par les points A et B.
    return((yB - yA) / (xB - xA))

def affine_cd(a, xA, yA) : # Retourne la fonction affine de coefficient directeur A et passant par le point B.
    b = yA - a*xA
    return(a*x + b)

def trace(xA, yA, xB, yB) : plt.plot([xA, xB], [yA, yB], color="black") # Trace la droite de A  B.

Le drapeau Anglais est un peu plus difficile (euphmisme). Replaons nous en dimensions 9x15. On dtermine :

  • Le coefficient directeur des droites diagonales montantes a. Celui des droites diagonales descendantes est donc -a.
  • L'paisseur des bandes rouges diagonales b, que l'on choisit de sorte ce que la premire bande infrieure droite s'arrte en un point donn (3.75 ; 5.5) selon le coefficient a.

plt.figure(figsize=(10, 5), dpi=80)

def od(c, d, e): return(e - c*d)
def point(c, d, e): return((e-d)/c)

x = np.linspace(0, 15.1, 150)
a = 1.8
b = od(a, 3.75, 5.5)

for m, r, f in [a, 3.75, x<3.75], [-a, 5.25, x>=5.25] :  
    y = m*x + od(m, r, 5) ; plt.fill_between(x[f], y[f], color="navy") # Triangles bleus infrieurs  
    y = -m*x + od(-m, r, 10) ; plt.fill_between(x[f], y[f], 15, color="navy") # Triangles bleus suprieurs

p = point(-a, 13, 9.5)
p2 = 9-p
    
for m, r, t, f, g in [-a, 13, od(-a, p2, 5.5), x<p, x>=p2], [a, od(a, 9-p, 9.5), 2, x>=9-p, x<p] : 
    y = m*x + r ; plt.fill_between(x[f], 9.5, y[f], color="navy") # Triangles milieu haut
    y = m*x + t ; plt.fill_between(x[g], y[g], 5.5, color="navy") # Triangles milieu bas

y1 = a*x + b ; y2 = y1 - b # Bandes rouges diagonales infrieure gauche & suprieure droite
p = point(a, b, 5.5)
plt.fill_between(x[(x<p) | (x>=5.25)], y1[(x<p) | (x>=5.25)], y2[(x<p) | (x>=5.25)], color="red")

p = point(-a, 15, 9.5)

for r, f in [15+b, x<p], [od(-a, 9, 0), x>=5.5] : # Bandes rouges diagonales suprieure gauche & infrieure droite
    y1 = -a*x + r ; y2 = y1 - b
    plt.fill_between(x[f], y1[f], y2[f], color="red")    

for i, j in [5.5, 6], [9, 9.5] : plt.fill_between(x, i, j, color="white") # Bandes blanches couvrantes

plt.fill_between([4, 5], [0, 0], [15, 15], color="red") # Bande rouge centrale verticale
plt.fill_between([0, 9], [6, 6], [9, 9], color="red") # Bande rouge centrale horizontale

plt.title("Union Jack", fontsize=35, pad=15)
